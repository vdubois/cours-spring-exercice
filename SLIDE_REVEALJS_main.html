Exercice (3/3)

Implémenter la classe suivante à la racine de votre projet (à côté de celle contenant la méthode main), lancez la classe main et constatez le résultat dans la console.

@Component
public class MonLanceur implements CommandLineRunner {
    private ItemsService itemsService;

    @Autowired
    public MonLanceur(ItemsService itemsService) {
        this.itemsService = itemsService;
    }

    @Override
    public void run(String... args) throws Exception {
        itemsService.addItem(new Item("Test !!!"));
        itemsService.addItem(new Item("Test 2 !!!"));
        System.out.println(itemsService.findAllItems());
    }
}
